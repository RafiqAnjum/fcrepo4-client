107     /**
108      * Create an HTTP helper for the specified repository.  If fedoraUsername and fedoraPassword are not null, then
109      * they will be used to connect to the repository.
110      * @param repositoryURL Fedora base URL.
111      * @param fedoraUsername Fedora username
112      * @param fedoraPassword Fedora password
113      * @param readOnly If true, throw an exception when an update is attempted.
114     **/
115     public HttpHelper(final String repositoryURL, final String fedoraUsername, final String fedoraPassword,
116                       final boolean readOnly) {
117         this.repositoryURL = repositoryURL;
118         this.readOnly = readOnly;
119 
120         final PoolingClientConnectionManager connMann = new PoolingClientConnectionManager();
121         connMann.setMaxTotal(MAX_VALUE);
122         connMann.setDefaultMaxPerRoute(MAX_VALUE);
123 
124         final DefaultHttpClient httpClient = new DefaultHttpClient(connMann);
125         httpClient.setRedirectStrategy(new DefaultRedirectStrategy());
126         httpClient.setHttpRequestRetryHandler(new StandardHttpRequestRetryHandler(0, false));
127 
128         // If the Fedora instance requires authentication, set it up here
129         if (!isBlank(fedoraUsername) && !isBlank(fedoraPassword)) {
130             LOGGER.debug("Adding BASIC credentials to client for repo requests.");
131 
132             final URI fedoraUri = URI.create(repositoryURL);
133             final CredentialsProvider credsProvider = new BasicCredentialsProvider();
134             credsProvider.setCredentials(new AuthScope(fedoraUri.getHost(), fedoraUri.getPort()),
135                                          new UsernamePasswordCredentials(fedoraUsername, fedoraPassword));
136 
137             httpClient.setCredentialsProvider(credsProvider);
138         }
139 
140         this.httpClient = httpClient;
141     }

143     /**
144      * Execute a request for a subclass.
145     **/
146     public HttpResponse execute( final HttpUriRequest request ) throws IOException, ReadOnlyException {
147         if ( readOnly ) {
148             switch ( request.getMethod().toLowerCase() ) {
149                 case "copy": case "delete": case "move": case "patch": case "post": case "put":
150                     throw new ReadOnlyException();
151                 default:
152                     break;
153             }
154         }
155 
156         return httpClient.execute(request);
157     }

159     /**
160      * Encode URL parameters as a query string.
161      * @param params Query parameters
162     **/
163     private static String queryString( final Map<String, List<String>> params ) {
164         final StringBuilder builder = new StringBuilder();
165         if (params != null && params.size() > 0) {
166             for (final Iterator<String> it = params.keySet().iterator(); it.hasNext(); ) {
167                 final String key = it.next();
168                 final List<String> values = params.get(key);
169                 for (final String value : values) {
170                     try {
171                         builder.append(key + "=" + URLEncoder.encode(value, "UTF-8") + "&");
172                     } catch (final UnsupportedEncodingException e) {
173                         builder.append(key + "=" + value + "&");
174                     }
175                 }
176             }
177             return builder.length() > 0 ? "?" + builder.substring(0, builder.length() - 1) : "";
178         }
179         return "";
180     }

209         }

211         final HttpPatch patch = new HttpPatch(repositoryURL + path);

213         patch.setHeader("Content-Type", contentTypeSPARQLUpdate);

237     /**
238      * Create a request to create/update content.
239      * @param path The datastream path.
240      * @param content Content parameters.
241     **/
242     public HttpPut createContentPutMethod(final String path, final Map<String, List<String>> params,
243                                           final FedoraContent content ) {
244         String contentPath = path;
245         if ( content != null && content.getChecksum() != null ) {
246             contentPath += "?checksum=" + content.getChecksum();
247         }
248 
249         final HttpPut put = createPutMethod( contentPath, params );
250 
251         // content stream
252         if ( content != null ) {
253             put.setEntity( new InputStreamEntity(content.getContent()) );
254         }
255 
256         // filename
257         if ( content != null && content.getFilename() != null ) {
258             put.setHeader("Content-Disposition", "attachment; filename=\"" + content.getFilename() + "\"" );
259         }
260 
261         // content type
262         if ( content != null && content.getContentType() != null ) {
263             put.setHeader("Content-Type", content.getContentType() );
264         }
265 
266         return put;
267     }

290     /**
291      * Retrieve RDF from the repository and update the properties of a resource
292      * @param resource The resource to update
293      * @return the updated resource
294     **/
295     public FedoraResourceImpl loadProperties( final FedoraResourceImpl resource ) throws FedoraException {
296         final String path = resource.getPath() + (resource instanceof FedoraDatastream ? "/fcr:metadata" : "");
297         final HttpGet get = createGetMethod(path, null);
298         if (resource instanceof FedoraObject) {
299             get.addHeader("Prefer", "return=representation; "
300                 + "include=\"http://fedora.info/definitions/v4/repository#EmbedResources\"");
301         }
302 
303         try {
304             get.setHeader("accept", "application/rdf+xml");
305             final HttpResponse response = execute(get);
306 
307             final String uri = get.getURI().toString();
308             final StatusLine status = response.getStatusLine();
309 
310             if (status.getStatusCode() == SC_OK) {
311                 LOGGER.debug("Updated properties for resource {}", uri);
312 
313                 // header processing
314                 final Header[] etagHeader = response.getHeaders("ETag");
315                 if (etagHeader != null && etagHeader.length > 0) {
316                     resource.setEtagValue( etagHeader[0].getValue() );
317                 }
318 
319                 // StreamRdf
320                 final HttpEntity entity = response.getEntity();
321                 final Lang lang = RDFLanguages.contentTypeToLang(entity.getContentType().getValue().split(":")[0]);
322                 final CollectorStreamTriples streamTriples = new CollectorStreamTriples();
323                 RiotReader.parse(entity.getContent(), lang, uri, streamTriples);
324                 resource.setGraph( RDFSinkFilter.filterTriples(streamTriples.getCollected().iterator(), Node.ANY) );
325                 return resource;
326             } else if (status.getStatusCode() == SC_FORBIDDEN) {
327                 LOGGER.info("request for resource {} is not authorized.", uri);
328                 throw new ForbiddenException("request for resource " + uri + " is not authorized.");
329             } else if (status.getStatusCode() == SC_BAD_REQUEST) {
330                 LOGGER.info("server does not support metadata type application/rdf+xml for resource {} " +
331                                      " cannot retrieve", uri);
332                 throw new BadRequestException("server does not support the request metadata type for resource " + uri);
333             } else if (status.getStatusCode() == SC_NOT_FOUND) {
334                 LOGGER.info("resource {} does not exist, cannot retrieve", uri);
335                 throw new NotFoundException("resource " + uri + " does not exist, cannot retrieve");
336             } else {
337                 LOGGER.info("unexpected status code ({}) when retrieving resource {}", status.getStatusCode(), uri);
338                 throw new FedoraException("error retrieving resource " + uri + ": " + status.getStatusCode() + " "
339                                           + status.getReasonPhrase());
340             }
341         } catch (final FedoraException e) {
342             throw e;
343         } catch (final Exception e) {
344             e.printStackTrace();
345             LOGGER.info("could not encode URI parameter", e);
346             throw new FedoraException(e);
347         } finally {
348             get.releaseConnection();
349         }
350     }
